371  // requires assembler really
372  volatile unsigned char zeroReg=0;
373  volatile unsigned int multipliedCounter=oscillators[c].phaseStep;
374  asm volatile
375  (
376  // high unsigned chars mult together = high unsigned char
377  "ldi %A[outVal],0" "\n\t"
378  "mul %B[phaseStep],%B[pitchB}]" "\n\t"
379  "mov %B[outVal],r0" "\n\t"
380  // ignore overflow into r1 (should never overflow)

382  "mul %A[phaseStep],%B[pitchB}]" "\n\t"
383  "add %A[outVal],r0" "\n\t"
384  // carry into high unsigned char
385  "adc %B[outVal],r1" "\n\t"
386  // high unsigned char* low unsigned char -> both unsigned chars
387  "mul %B[phaseStep],%A[pitchB}]" "\n\t"
388  "add %A[outVal],r0" "\n\t"
389  // carry into high unsigned char

391  // low unsigned char * low unsigned char -> round
