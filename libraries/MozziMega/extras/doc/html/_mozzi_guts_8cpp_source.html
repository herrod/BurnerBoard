30 /*
31 ATmega328 technical manual, Section 12.7.4:
32 The dual-slope operation [of phase correct pwm] has lower maximum operation
33 frequency than single slope operation. However, due to the symmetric feature
34 of the dual-slope PWM modes, these modes are preferred for motor control
35 applications.
36 Due to the single-slope operation, the operating frequency of the
37 fast PWM mode can be twice as high as the phase correct PWM mode that use
38 dual-slope operation. This high frequency makes the fast PWM mode well suited
39 for power regulation, rectification, and DAC applications. High frequency allows
40 physically small sized external components (coils, capacitors)..
41 
42 DAC, that's us! Fast PWM.
43 
44 PWM frequency tests
45 62500Hz, single 8 or dual 16 bits, bad aliasing
46 125000Hz dual 14 bits, sweet
47 250000Hz dual 12 bits, gritty, if you're gonna have 2 pins, have 14 bits
48 500000Hz dual 10 bits, grittier
49 16384Hz single nearly 9 bits (original mode) not bad for a single pin, but carrier freq noise can be an issue
50 */

150 void audioHook() // 2us excluding updateAudio()
151 {
152  static unsigned long output_buffer_head = 0;
153  long output_gap = output_buffer_head - output_buffer_tail; // wraps to a big number if it's negative, and will take a long time to wrap
154 
155 #if USE_AUDIO_INPUT // 3us
156 
157  static unsigned long input_buffer_tail =0;
158  input_gap = input_buffer_head - input_buffer_tail; // wraps to a big number if it's negative, and will take a long time to wrap
159  if ((output_gap < BUFFER_NUM_CELLS) && do_update_audio) {
160  input_buffer_tail++;
161  audio_input = input_buffer[input_buffer_tail & (BUFFER_NUM_CELLS-1)];
162 
163 #else
164 
165  if(output_gap < BUFFER_NUM_CELLS) // prevent writing over cells which haven't been output yet
166  {
167 
168 #endif
169  output_buffer_head++;
170  output_buffer[(unsigned char)output_buffer_head & (unsigned char)(BUFFER_NUM_CELLS-1)] = (unsigned int) (updateAudio() + AUDIO_BIAS);
171  }

317  /* Sets up Timer 0 for control interrupts. This is the same for all output
318  options Using Timer0 for control disables Arduino's time functions but also
319  saves on the interrupts and blocking action of those functions. May add a config
320  option for Using Timer2 instead if needed. (MozziTimer2 can be re-introduced for
321  that). */

327  pre_mozzi_OCR0A = OCR0A;
